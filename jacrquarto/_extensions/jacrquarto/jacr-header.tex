% ---- Tables ----
% JACR asks authors to avoid shading in tables (incompatible with editing
% software). booktabs rules provide clean, unshaded tables by default.
% Unlike JCR, JACR does not mandate a sans-serif table font, so tables keep
% the body (Times) font.

% ---- Captions ----
\usepackage[labelsep=newline,labelfont={bf},textfont={}]{caption}

% ---- MANUSCRIPT SETUP ----
% JACR heading scheme (Guidelines for Authors, "Section and paragraph
% headings"):
%   Major headings:  flush left, capitalized, bold.
%   Subheadings:     mixed case, flush left, italic.
%   Tertiary:        mixed case, flush left, bold, run-in, ends with a period.
% JACR requires pages to be numbered. Spacing: the accepted-manuscript file
% is double-spaced; for the review submission JACR encourages 1.5 spacing.
% The default here is double (safe for both); override with linestretch: 1.5.
\newcommand{\jacrsetupmanuscript}{%
  \pagestyle{plain}%
  \raggedright%
  \setlength{\parindent}{0.5in}%
  \titleformat{\section}%
    {\normalfont\normalsize\bfseries\raggedright\MakeUppercase}{}{0pt}{}%
  \titlespacing*{\section}{0pt}{\baselineskip}{\baselineskip}%
  \titleformat{\subsection}%
    {\normalfont\normalsize\itshape\raggedright}{}{0pt}{}%
  \titlespacing*{\subsection}{0pt}{\baselineskip}{0.5\baselineskip}%
  \titleformat{\subsubsection}[runin]%
    {\normalfont\normalsize\bfseries}{}{0pt}{}[.]%
  \titlespacing*{\subsubsection}{\parindent}{\baselineskip}{0.5em}%
}

% ---- ARTICLE SETUP ----
% Publication-style single-column layout.
\newcommand{\jacrsetuparticle}{%
  \pagestyle{fancy}%
  \fancyhf{}%
  \fancyfoot[C]{\small\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
  \fancypagestyle{plain}{%
    \fancyhf{}%
    \fancyfoot[C]{\small\thepage}%
    \renewcommand{\headrulewidth}{0pt}%
  }%
  \thispagestyle{plain}%
  \setlength{\parindent}{1.5em}%
  \titleformat{\section}%
    {\normalfont\large\bfseries\raggedright}{}{0pt}{}%
  \titlespacing*{\section}{0pt}{18pt}{6pt}%
  \titleformat{\subsection}%
    {\normalfont\normalsize\bfseries\itshape\raggedright}{}{0pt}{}%
  \titlespacing*{\subsection}{0pt}{14pt}{4pt}%
  \titleformat{\subsubsection}%
    {\normalfont\normalsize\itshape\raggedright}{}{0pt}{}%
  \titlespacing*{\subsubsection}{0pt}{10pt}{3pt}%
  \captionsetup{font=small,labelfont={bf},textfont={}}%
}
